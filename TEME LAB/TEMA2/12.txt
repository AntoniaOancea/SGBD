WHERE
EXTRACT(MONTH FROM BOOK_DATE) = EXTRACT(MONTH FROM SYSDATE)
AND EXTRACT(YEAR FROM BOOK_DATE)=EXTRACT(YEAR FROM SYSDATE)
GROUP BY BOOK_DATE;
